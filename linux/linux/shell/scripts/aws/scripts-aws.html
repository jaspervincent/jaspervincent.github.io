cat ec2.log |jq '.Reservations[].Instances[]|[ ([.Tags[]?|(if .Key == "Name" then .Value  else ""  end )]|join("")), .InstanceId, .PrivateIpAddress, .KeyName, .VpcId, .SubnetId, .State.Name, ([(.Tags[]?|[(.Key, .Value)]|join("="))]|join("@"))] | join(",")'  | sed 's/"//g' |sed '1i \Name,InstanceId,PrivateIpAddress,KeyName,VpcId,SubnetId,State.Name,Tags(kv1@kv2)' > ec2.csv

生成 xls 文件 ec2.xls

# 重新优化标签
import numpy as np
import pandas as pd

source_file = "ec2.csv"
dist_file = "ec2.xls"

def read_csv(file: str) -> [str]:
    with open(file, "r") as f:
        data = f.readlines()
        return data
    return []


def write_excel(file: str, data):
    A = np.array(data)
    data = pd.DataFrame(A)

    writer = pd.ExcelWriter(file)  # 写入Excel文件
    data.to_excel(writer, 'page_1', )  # ‘page_1’是写入excel的sheet名
    writer.save()


def process():
    dist_data = []
    title = ['Name', 'InstanceId', 'PrivateIpAddress', 'KeyName', 'VpcId', 'SubnetId', 'State']
    data = read_csv(source_file)
    for index, row in enumerate(data):
        dist_row = [""] * 100
        if index == 0:
            dist_data.append(title)
            continue
        info = row.split(",")
        for index, i in enumerate(info):
            if index < 7:
                dist_row[index] = i
            if index == 7:
                for kvs in i.split("@"):
                    kv = kvs.split("=")
                    if len(kv) != 2:
                        continue
                    k = kv[0]
                    v = kv[1]
                    if k not in title:
                        title.append(k)
                    dist_row[title.index(k)] = v
        dist_data.append(dist_row)
    write_excel(dist_file, dist_data)


if __name__ == '__main__':
    process()

#!/bin/env bash
#
set -eux;
__check_system() {
  OS_RELEASE="$(awk -F'=' '/^NAME/{print $2}' /etc/os-release| sed 's@\"@@g')" `#OS: Amazon Linux|CentOS Linux|Ubuntu`
  ARCH="$(arch)" `#ARCH: x86_64(amd)|aarch64(arm)`
}

__conf() {
  __check_system
  echo "#------Initial ${server} configuration----------"
  URL="${URL}"
  [[ "$OS_RELEASE" =~ "Linux" ]] && INSTALL=([cmd]='rpm' [arg]='-ivh' [os]="centos" [type]='rpm') ||:
  [[ "$OS_RELEASE" =~ "Ubuntu" ]] && INSTALL=([cmd]='dpkg' [arg]='--install' [os]="ubuntu" [type]='deb')||:
  [ -z "${INSTALL[cmd]}" ] && { echo Unable to determine linux distribution version; exit 1; }||:
  case "$ARCH" in
    'x86_64')
    INSTALL['pkg']="${pkg_amd}"
    downloadUrl="${URL}/${INSTALL[pkg]}"
    sleep 1
    ;;
    'aarch64')
    INSTALL['pkg']="${pkg_amd}"
    downloadUrl="${URL}/${INSTALL[pkg]}"
    sleep 1
    ;;
    *) echo >&2 "error: unsupported architecture: '$ARCH'"; exit 1 ;;
  esac
}

_node_exporter() {
  echo "#------Check ${server}----------"
  if pgrep  ${server}  &>/dev/null; then echo ${server} is installed ; exit 0; fi

  local pkg_amd=node_exporter-1.5.0.linux-amd64.tar.gz
  local pkg_arm="node_exporter-1.5.0.linux-arm64.tar.gz"
  local URL="https://github.com/prometheus/node_exporter/releases/download/v1.5.0"
  __conf

  echo "#------Install ${server}----------"
  curl -s -L -o "${INSTALL[pkg]}" "$downloadUrl" \
    && tar -zxvf "${INSTALL[pkg]}"  --strip-components=1 -C /usr/local/bin ${INSTALL[pkg]%%.tar.gz*}/node_exporter \
    && rm -f "${INSTALL[pkg]}"
  sudo tee /etc/systemd/system/node_exporter.service <<"EOF"
[Unit]
Description=Node Exporter

[Service]
User=root
Type=simple
ExecStart=/usr/local/bin/node_exporter

[Install]
WantedBy=multi-user.target
EOF

  sudo systemctl daemon-reload \
    && sudo systemctl restart node_exporter \
    && sudo systemctl status node_exporter \
    && sudo systemctl enable node_exporter
}

_serverlist() {
  SERVICE+=("node_exporter")
}

install_server() {
  unset INSTALL
  declare -A INSTALL
  local OS_RELEASE  ARCH server
  local SERVICE=()
  _serverlist
  for server in ${SERVICE[@]};do
  {
    _$server ; sleep 1;
  }
  done
}

install_server

  [[ "$OS_RELEASE" =~ "Ubuntu" ]] && INSTALL=([cmd]='dpkg' [arg]='--install' [os]="ubuntu" [type]='deb')||:
  local pkg_amd="QualysCloudAgent-${INSTALL[os]}-${ENV}.${INSTALL[type]}"
  local pkg_arm="QualysCloudAgent-${INSTALL[os]}-arm-${ENV}.${INSTALL[type]}"
  __conf

  echo "------Install ${server}----------"
  wget "$downloadUrl" && \
  sudo "${INSTALL[cmd]}" "${INSTALL[arg]}" "${INSTALL[pkg]}" \
    && sudo /usr/local/qualys/cloud-agent/bin/qualys-cloud-agent.sh ActivationId=a23f2aa1-989d-46a2-925e-b0a02eb66de7 CustomerId=5866b668-2177-5153-8182-3fc0ae5fd420 \
    && sleep 2 \
    && sudo sed -ri '/\[Install\]/i\CPUQuota=5%\nMemoryLimit=250M' /lib/systemd/system/qualys-cloud-agent.service \
    && sudo systemctl daemon-reload \
    && sudo systemctl restart qualys-cloud-agent \
    && sudo systemctl status qualys-cloud-agent \
    && sleep 3 \
    && rm -f "${INSTALL[pkg]}"
}
