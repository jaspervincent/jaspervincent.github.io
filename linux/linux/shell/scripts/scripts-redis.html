统计前缀的key的数量

方法一：使用 DBSIZE + KEYS 命令（不推荐用于生产） 

# 危险！不要在正式环境使用
redis-cli KEYS "UA:ACCESS_TOKEN_CACHE:*" | wc -l
#KEYS 命令会扫描整个数据库，在数据量大的情况下会严重阻塞 Redis，导致服务不可用

方法二：使用 SCAN 命令（推荐，安全）

SCAN 命令使用游标方式分批遍历，不会阻塞 Redis 服务。

一次性统计： 

redis-cli --raw SCAN 0 MATCH "UA:ACCESS_TOKEN_CACHE:*" COUNT 1000 | grep -v "^[0-9]" | grep -v "^$" | wc -l

脚本 

#!/bin/bash

REDIS_HOST="localhost"
REDIS_PORT=6379
PATTERN="UA:ACCESS_TOKEN_CACHE:*"

total_count=0
cursor=0

while true; do
  # 执行 SCAN 命令
  result=$(redis-cli -h $REDIS_HOST -p $REDIS_PORT --raw SCAN $cursor MATCH "$PATTERN" COUNT 1000)

  # 提取游标和 keys
  cursor=$(echo "$result" | head -n 1)
  keys=$(echo "$result" | tail -n +2 | grep -v "^$")

  # 统计当前批次的 key 数量
  if [ -n "$keys" ]; then
    count=$(echo "$keys" | wc -l)
    total_count=$((total_count + count))
    echo "Processed batch: $count keys (Total so far: $total_count)"
  fi

  # 如果游标为 0，结束循环
  if [ "$cursor" -eq "0" ]; then
    break
  fi
done

echo "=========================================="
echo "Total keys with pattern '$PATTERN': $total_count"

方法三：使用 Lua 脚本（最高效）

count_keys.lua 

local pattern = ARGV[1]
local cursor = 0
local count = 0

repeat
    local res = redis.call('SCAN', cursor, 'MATCH', pattern, 'COUNT', 1000)
    cursor = tonumber(res[1])
    local keys = res[2]
    count = count + #keys
until cursor == 0

return count

执行脚本： 

redis-cli --eval count_keys.lua , "UA:ACCESS_TOKEN_CACHE:*"

查看key内存占用

方法一：使用 Redis 自带的命令组合（推荐用于少量 Key） 

#!/bin/bash

REDIS_HOST="localhost"
REDIS_PORT=6379
KEY_PATTERN="UA:ACCESS_TOKEN_CACHE:*"

cli_cmd="redis-cli -h $REDIS_HOST -p $REDIS_PORT --raw"

# 将匹配的 key 读入一个数组
# 使用 SCAN 多次迭代以确保获取所有 key
keys=()
cursor=0
while true; do
  # 执行 SCAN 命令
  reply=$($cli_cmd SCAN $cursor MATCH "$KEY_PATTERN" COUNT 1000)
  # 提取新的游标值（第一行）
  cursor=$(echo "$reply" | head -n 1)
  # 提取 key 列表（从第二行开始），并过滤空行和换行符
  new_keys=$(echo "$reply" | tail -n +2 | tr -d '\r' | grep -v '^$')

  # 如果有关键词，则读入数组
  if [ -n "$new_keys" ]; then
    while IFS= read -r key; do
      if [ -n "$key" ]; then
        keys+=("$key")
      fi
    done <<< "$new_keys"
  fi

  # 如果游标为 0，表示迭代结束
  if [ "$cursor" -eq 0 ]; then
    break
  fi
done

total_memory=0
key_count=0
echo "Found ${#keys[@]} keys with pattern '$KEY_PATTERN'"
echo "Starting memory analysis..."
echo "------------------------------------------"

for key in "${keys[@]}"; do
  memory_usage=$($cli_cmd MEMORY USAGE "$key" 2>/dev/null)
  if [[ $memory_usage =~ ^[0-9]+$ ]]; then
    echo "Key: $key, Memory: $memory_usage bytes"
    total_memory=$((total_memory + memory_usage))
    key_count=$((key_count + 1))
  else
    echo "Warning: Could not get memory for key '$key' (returned: $memory_usage)"
  fi
done

echo "=========================================="
echo "Summary:"
echo "Keys found: ${#keys[@]}"
echo "Keys successfully processed: $key_count"
echo "Total memory usage:"
echo "Bytes: $total_memory"
echo "KB: $((total_memory / 1024))"
echo "MB: $(( (total_memory + 524288) / 1048576 ))" # 更精确的四舍五入

方法2：使用 Lua 脚本（高效，原子性执行） 

local pattern = ARGV[1] -- 从参数获取模式，例如 'UA:ACCESS_TOKEN_CACHE:*'
local cursor = 0
local total_memory = 0
local key_count = 0

repeat
    -- 使用 SCAN 命令
    local res = redis.call('SCAN', cursor, 'MATCH', pattern, 'COUNT', 1000)
    cursor = tonumber(res[1])
    local keys = res[2]

    -- 遍历当前批次的 key
    for i, key in ipairs(keys) do
        -- 获取每个 key 的内存使用量
        local mem = redis.call('MEMORY', 'USAGE', key)
        total_memory = total_memory + mem
        key_count = key_count + 1
        -- 如果想看每个 key 的详情，可以取消下面的注释
        -- redis.log(redis.LOG_NOTICE, key .. ": " .. mem .. " bytes")
    end
until cursor == 0

-- 返回结果：处理了的 key 数量和总内存占用（字节）
return {key_count, total_memory}

方法三：使用 redis-rdb-tools（最准确，离线分析） 

#1 生成 RDB 快照
redis-cli SAVE # 或者 BGSAVE

#2 使用 rdb 命令分析
# 生成一个 CSV 报告，只包含匹配前缀的 Key
rdb -c memory /path/to/dump.rdb --bytes 0 -f memory_report.csv

# 然后使用 grep 和 awk 过滤出你需要的 prefix 并求和
grep "^UA:ACCESS_TOKEN_CACHE:" memory_report.csv | awk -F, '{sum += $4} END {print "Total Memory:", sum, "bytes", "(" sum/1024/1024 "MB)"}'

#或者一步到位：
rdb -c memory /path/to/dump.rdb | grep "UA:ACCESS_TOKEN_CACHE:" | awk -F, '{sum += $4} END {print "Total Memory:", sum, "bytes", "(" sum/1024/1024 "MB)"}'
