#!/bin/bash

KEY='DB11_BAOMAN_COMMEN DB11_BAOMAN_FRAME DB11_BAOMAN_LOCAL DB11_BAOMAN_NOTE DB11_BAOMAN_PRAISE'
#
for k in $KEY ; do
/usr/bin/redis-cli -h r-redis.rds.aliyuncs.com -p 6379 -a MANDY0611kir -n 11 keys "$k*" |xargs /usr/bin/redis-cli -h redis.rds.aliyuncs.com -p 6379 -a MANDY0611kir -n 11  del
sleep 2
done
