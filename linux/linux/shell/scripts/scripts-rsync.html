    then
         echo start deploy for staging env
         staging_deploy
    elif [ "${BUILD_TYPE}" = "prod" ]
    then
        echo start deploy for production env;
        prod_deploy
    elif [ "${BUILD_TYPE}" = "master" ]
    then
        echo start deploy for production env;
        master_deploy
    elif [ "${BUILD_TYPE}" = "privacy" ]
    then
        echo start deploy for privacy env;
        privacy_deploy
    fi
    releaselock deploy.lock
}
