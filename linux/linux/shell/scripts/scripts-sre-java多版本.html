安装 java

tar xf jdk-8u351-linux-aarch64.tar.gz -C /usr/local/
ln -sv /usr/local/jdk1.8.0_351/  /usr/local/java

cat /etc/profile.d/java.sh
export JAVA_HOME=/usr/local/java
export JRE_HOME=/usr/local/java/jre
export CLASSPATH=.:$JAVA_HOME/lib/dt.jar:/lib/tools.jar:$JRE_HOME/lib:$CLASSPATH
export PATH=$JAVA_HOME/bin:$PATH

scripts-sre-java 多版本

默认 jdk8，使用 jdk8, jdk11 命令切换环境 

[root@ip-10-204-3-118 ~]# cat /etc/profile.d/os_env.sh
JAVA_8_HOME="/usr/local/jdk1.8.0_351"
JAVA_11_HOME="/usr/local/jdk-11.0.18"
JRE_8_HOME=/usr/local/jdk1.8.0_351/jre
JRE_11_HOME=/usr/local/jdk-11.0.18/jre

alias jdk8="export JAVA_HOME=$JAVA_8_HOME  JRE_HOME=$JRE_8_HOME  PATH=$JAVA_8_HOME/bin:$PATH"
alias jdk11="export JAVA_HOME=$JAVA_11_HOME JRE_HOME=$JRE_11_HOME PATH=$JAVA_11_HOME/bin:$PATH "

JAVA_HOME=$JAVA_8_HOME
JRE_HOME=$JAVA_8_HOME/jre

export JAVA_HOME=/usr/local/java
export CLASSPATH=.:$JAVA_HOME/lib/dt.jar:/lib/tools.jar:$JRE_HOME/lib:$CLASSPATH
export PATH=$JAVA_HOME/bin:$PATH

[root@ip-10-204-3-118 ~]# java -version

Java HotSpot(TM) 64-Bit Server VM (build 25.351-b10, mixed mode)
[root@ip-10-204-3-118 ~]# jdk11
