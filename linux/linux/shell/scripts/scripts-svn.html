创建svn账号


#!/bin/bash
# File: svn-user
# Usage:  ./svn-user

user_file="/data/svn/conf/passwd"

if [ ! -f $user_file ]; then
    echo "Error certificate file does not exist"
    exit 1
fi

if [ -z $1 ]; then
    read -p Enter add user name: $user_name
else
    user_name=$1
fi

for file_user in `awk '{print $1}' $user_file`; do
    if [ $file_user -eq $user_name ]; then
        echo "$user_name user already exists"
        exit 1
    fi
done

user_passwd=`openssl rand -hex 16`

echo "$user_name = $user_passwd" >$user_file
cat << END
===================================================================

                       SVN add user success
                        User name: $user_name
         User password: $user_passwd

===================================================================
END
